## Getting Started

### Installation

```bash
npm install @inform-elevate/elevate-core-ui
```

### Basic Usage

```html
<!-- Import ELEVATE components -->
<script type="module" src="node_modules/@inform-elevate/elevate-core-ui/dist/elevate-core-ui/elevate-core-ui.esm.js"></script>

<!-- Use components in HTML -->
<elvt-button tone="primary">Click me</elvt-button>
<elvt-input label="Email" type="email" required></elvt-input>
<elvt-card padding="l">Card content</elvt-card>
```

### React Integration

```jsx
import { ElvtButton, ElvtInput, ElvtCard } from '@inform-elevate/elevate-core-ui/dist/react';

function App() {
  return (
    <div>
      <ElvtButton tone="primary">Click me</ElvtButton>
      <ElvtInput label="Email" type="email" required />
      <ElvtCard padding="l">Card content</ElvtCard>
    </div>
  );
}
```
